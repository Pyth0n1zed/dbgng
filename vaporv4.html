Open the Ghost Tab const ghost = window.open('about:blank', '_blank'); if (ghost) { // 2. Inject Anura into the Ghost const loader = gDoc.createElement('script'); loader.src = ANURA_SRC; loader.id = "rspa-core-sync"; gDoc.body.appendChild(loader); // 4.
